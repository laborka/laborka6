#include<iostream>
#include<conio.h>
#include<string>

using namespace std;

string usun_spacje(string tekst) {

	int ii = 0, n;

	string nowy;


	n = tekst.length();
	for (ii = 0; ii < n; ii++)
		if (tekst[ii] != ' ')
			nowy += tekst[ii]; //dodawanie stringow
	return nowy;
}



string odwroc(string tekst)
{
	int i = 0;
	int n;
	string kot;
	n = tekst.length();

	for (int i = 0; i < n; i++)
	{
		kot.insert(i, tekst, n - i - 1, 1); //miejsce wstawienia,tekst,co z tekstu,ile
	}


	return kot;



}



int main() {

	cout << "usun spacje:" << endl;

	getline(cin, tekst);

	cout << usun_spacje(tekst);

	string tekst2;
